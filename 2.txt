int button=7;
int diod[3]={3,5,6};
int rd[5]={255,255,255,75,255};
int gd[5]={255,182,222,0,0};
int bd[5]={255,193,173,130,255};
bool rt=false;

void setup() {
  pinMode(button,INPUT);
  for (int i=0;i<3; i++)
  pinMode(diod[i],OUTPUT);

}
int i=0;
void loop() {
 if (digitalRead(button)==1){
  digitalWrite(diod[0], rd[i]);
  digitalWrite(diod[1], gd[i]);
  digitalWrite(diod[2], bd[i]);
 delay(1000); 
 i++;
 if(i==5)i=0;
}

}
